CALCULO DA IDADE
programa {
	
	inteiro idadePessoas
	inteiro idadeMeses
	inteiro idadeDias
	inteiro idadeHoras
	inteiro idadeMinutos
	
	
	funcao inicio() {
		escreva("Qual È a sua idade?")
		leia (idadePessoas)
		
		idadeMeses = idadePessoas * 12
		idadeDias = idadePessoas * 365
		idadeHoras = idadeDias * 24
		idadeMinutos = idadeHoras * 60
		
		escreva("Sua idade em meses È: " + idadeMeses + "\n")
		escreva("Sua idade em dias È: " + idadeDias + "\n")
		escreva("Sua idade em horas È: " + idadeHoras + "\n")
		escreva("Sua idade em horas È: " + idadeMinutos)
	}
}
